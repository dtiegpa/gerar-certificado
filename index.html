// Novo sistema baseado em Google Slides e Google Sheets - Gerador de Certificados via Apps Script /** * Script para Google Apps Script * Substitui campos em um modelo do Google Slides e gera PDFs para cada aluno */ function gerarCertificadosDoSlides() { const modeloId = '1nGgazLyGhSvyo_6Vc2lkloSFc3W90wcG8bBCaPaFMrI'; const pastaDestino = DriveApp.getFolderById('1-z6kiofA3bnOMTEjbF8mBzzCxyNVFzKR'); const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Certificados'); const dados = sheet.getDataRange().getValues(); const cabecalho = dados[0]; for (let i = 1; i < dados.length; i++) { const linha = dados[i]; const info = {}; cabecalho.forEach((col, idx) => info[col.trim()] = linha[idx]); const copia = DriveApp.getFileById(modeloId).makeCopy(`Certificado - ${info.NOME}`, pastaDestino); const slide = SlidesApp.openById(copia.getId()); slide.getSlides().forEach(pagina => { cabecalho.forEach(col => { const marcador = `{{${col.trim()}}}`; pagina.replaceAllText(marcador, info[col.trim()]); }); }); slide.saveAndClose(); const pdf = DriveApp.getFileById(copia.getId()).getAs('application/pdf'); pastaDestino.createFile(pdf); DriveApp.getFileById(copia.getId()).setTrashed(true); } }
